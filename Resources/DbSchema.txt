CREATE TABLE TimeEntries (
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	Date TEXT NOT NULL,
	ChargeCode TEXT,
	StartTime REAL,
	EndTime REAL,
	Comment TEXT	
);

CREATE TABLE ChargeCodes (
	ChargeCode TEXT PRIMARY KEY,
	Description TEXT,
	ID INTEGER AUTOINCREMENT
);

CREATE INDEX TimeEntries_Date ON TimeEntries(Date);
